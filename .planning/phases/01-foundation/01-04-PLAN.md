**1. `src/middleware.ts`** — Combined Clerk auth + next-intl locale middleware. This is the MOST CRITICAL integration point:

```typescript
import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server'
import createMiddleware from 'next-intl/middleware'
import { routing } from './i18n/routing'

const intlMiddleware = createMiddleware(routing)

// Protected routes — use /:locale pattern to handle both prefixed and unprefixed
const isProtectedRoute = createRouteMatcher([
  '/:locale/dashboard(.*)',
  '/:locale/settings(.*)',
  '/:locale/capture(.*)',
  '/dashboard(.*)',
  '/settings(.*)',
  '/capture(.*)',
])

// Public routes that should not require auth
const isPublicRoute = createRouteMatcher([
  '/:locale/auth(.*)',
  '/auth(.*)',
  '/:locale/sso-callback(.*)',
  '/sso-callback(.*)',
])

export default clerkMiddleware(async (auth, req) => {
  if (isProtectedRoute(req)) {
    await auth.protect()
  }
  return intlMiddleware(req)
})

export const config = {
  matcher: [
    // Skip Next.js internals and static files
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    '/(api|trpc)(.*)',
  ],
}
```

CRITICAL PITFALL (from RESEARCH.md): Use `'/:locale/dashboard(.*)'` pattern AND the unprefixed `'/dashboard(.*)'` pattern. With `localePrefix: 'as-needed'`, the default locale (English) has NO prefix, so route matchers need to handle both cases.

CRITICAL: This file MUST be named `middleware.ts` for Next.js 15. Next.js 16+ uses `proxy.ts` — do NOT rename it.

**2. `src/app/[locale]/(app)/layout.tsx`** — Protected app layout with sidebar:

```typescript
import { SidebarProvider, SidebarInset, SidebarTrigger } from '@/components/ui/sidebar'
import { AppSidebar } from '@/components/layout/app-sidebar'
import { setRequestLocale } from 'next-intl/server'

type Props = {
  children: React.ReactNode
  params: Promise<{ locale: string }>
}

export default async function AppLayout({ children, params }: Props) {
  const { locale } = await params
  setRequestLocale(locale)

  return (
    <SidebarProvider>
      <AppSidebar />
      <SidebarInset>
        <header className="flex h-14 items-center gap-2 border-b px-4 lg:px-6">
          <SidebarTrigger className="-ml-1" />
        </header>
        <main className="flex-1 p-4 lg:p-6">
          {children}
        </main>
      </SidebarInset>
    </SidebarProvider>
  )
}
```

The `(app)` route group wraps all authenticated pages (dashboard, settings, capture). Pages outside this group (auth, sso-callback) don't have the sidebar.
