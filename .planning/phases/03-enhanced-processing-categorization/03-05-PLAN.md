**1. Create `src/lib/actions/batch-upload.ts`:**

Server action for saving batch results:

```typescript
export const batchSaveAll = authActionClient
  .inputSchema(z.object({
    receiptIds: z.array(z.string()).min(1),
  }))
  .action(async ({ parsedInput: { receiptIds }, ctx: { userId } }) => {
    // Mark all specified receipts as verified (auto-accept AI extraction)
    // For each receipt: set verifiedAt = now() if extractionStatus = "completed"
    // Uses prisma.$transaction for atomicity
    const result = await prisma.$transaction(
      receiptIds.map(id =>
        prisma.capturedReceipt.updateMany({
          where: { id, userId, extractionStatus: "completed", verifiedAt: null },
          data: { verifiedAt: new Date() },
        })
      )
    );
    return { saved: result.reduce((sum, r) => sum + r.count, 0) };
  });
```
